  9 /**
 10     * The logistic function (sigmoid)
 11     * @param {Number} value The value
 12     * @param {Number} [midValue=0] The midpoint value, at which the function returns max/2
 13     * @param {Number} [min=0] The minimum return value
 14     * @param {Number} [max=1] The maximum return value
 15     * @param {Number} [rate=1] The logistic growth rate or steepness of the function
 16     * @return {Number} The result in the range [min, max] (excluding min and max)
 17     */
 18 FzMath.logistic = function( value, midValue, min, max, rate)
 19 {
 20     if (typeof midValue === "undefined") midValue = 0;
 21     if (typeof max === "undefined") max = 1;
 22     if (typeof min === "undefined") min = 0;
 23     if (typeof rate === "undefined") rate = 1;
 24     var exp = -rate*(value - midValue);
 25     var result = 1 / (1 + Math.pow(Math.E, exp));
 26     return result * (max-min) + min;
 27 }
 28 
 29 /**
 30     * The inverse logistic function (inverse sigmoid)
 31     * @param {Number} v The variable
 32     * @param {Number} [midValue=0] The midpoint value, at which the function returns max/2 in the original logistic function
 33     * @param {Number} [min=0] The minimum return value of the original logistic function
 34     * @param {Number} [max=1] The maximum return value of the original logistic function
 35     * @param {Number} [rate=1] The logistic growth rate or steepness of the original logistic function
 36     * @return {Number} The result
 37     */
 38 FzMath.inverseLogistic = function ( v, midValue, min, max, rate)
 39 {
 40     if (typeof midValue === "undefined") midValue = 0;
 41     if (typeof max === "undefined") max = 1;
 42     if (typeof min === "undefined") min = 0;
 43     if (typeof rate === "undefined") rate = 1;
 44 
 45     if (v == min) return 0;
 46     
 47     return midValue - Math.log( (max-min)/(v-min) - 1) / rate;
 48 }

 50 /**
 51     * The gaussian function
 52     * @param {Number} value The variable
 53     * @param {Number} [min=0] The minimum return value
 54     * @param {Number} [max=1] The maximum return value
 55     * @param {Number} [center=0] The center of the peak
 56     * @param {Number} [fwhm=1] The full width at half maximum of the curve
 57     * @return {Number} The result
 58     */
 59 FzMath.gaussian = function( value, min, max, center, fwhm)
 60 {
 61     if (typeof max === "undefined") max = 1;
 62     if (typeof min === "undefined") min = 0;
 63     if (typeof center === "undefined") center = 0;
 64     if (typeof fwhm === "undefined") fwhm = 1;
 65     if (fwhm === 0 && value == center) return max;
 66     else if (fwhm === 0) return 0;
 67 
 68     var exp = -4 * Math.LN2;
 69     exp *= Math.pow((value - center),2);
 70     exp *= 1/ Math.pow(fwhm, 2);
 71     var result = Math.pow(Math.E, exp);
 72     return result * (max-min) + min;
 73 }
 74 
 75 /**
 76     * The inverse gaussian function
 77     * @param {Number} v The variable
 78     * @param {Number} [min=0] The minimum return value of the corresponding gaussian function
 79     * @param {Number} [max=1] The maximum return value of the corresponding gaussian function
 80     * @param {Number} [center=0] The center of the peak of the corresponding gaussian function
 81     * @param {Number} [fwhm=1] The full width at half maximum of the curve of the corresponding gaussian function
 82     * @return {Number[]} The two possible results, the lower is the first in the list. If both are the same, it is the maximum
 83     */
 84 FzMath.inverseGaussian = function( v, min, max, center, fwhm)
 85 {
 86     if (typeof max === "undefined") max = 1;
 87     if (typeof min === "undefined") min = 0;
 88     if (typeof center === "undefined") center = 0;
 89     if (typeof fwhm === "undefined") fwhm = 1;
 90     if (v == 1) return [center, center];
 91     if (v === 0) return [center + fwhm/2, center - fwhm/2];
 92     if (fwhm === 0) return [center, center];
 93 
 94     var result = (v-min)/(max-min);
 95     result = Math.log( result ) * Math.pow(fwhm,2);
 96     result = result / ( -4 * Math.LN2 );
 97     result = Math.sqrt( result );
 98     return [ result + center, -result + center ];
 99 }

101 /**
102     * Returns the mean of a set of values
103     * @param {Number[]} values The values
104     * @return {Number} The mean
105     */
106 FzMath.mean = function( values )
107 {
108     var num = values.length;
109     var result = 0;
110     for (var i = 0; i < num; i++)
111     {
112         result += values[i];
113     }
114     return result / num;
115 }

119 /**
120     * Do not use the constructor of this class, use {@link FuzzyLogic.newSet} to create a new set.<br >
121     * Most of the time you won need to access the properties nor use the methods of this class, but use the methods of {@link FuzzyLogic}, {@link FuzzyValue}, {@link FuzzyVeracity}
122     * @class
123     * @classdesc A Fuzzy set.
124     * @property {string} name The name of this set .
125     * @property {Number} valueNOT One of the closest value which is not in the set (either above or below).
126     * @property {Number} valueIS The value which best fits in the set, the most extreme/maximum in the set.
127     * @property {FuzzyShape} shape The shape (i.e. interpolation or transition) when getting in (below) the set.
128     * @property {FuzzyShape} shapeAbove The shape (i.e. interpolation or transition) when getting out (above) of the set.
129     * @property {Number} plateauMin The value above which a value is considered completely included.
130     * @property {Number} plateauMax The value under which a value is considered completely included.
131     */
132 function FuzzySet( name, valueNot, valueIS, shape, shapeAbove, plateauMin, plateauMax, algorithm)
133 {
134     var min;
135     var max;
136     if (valueNot > valueIS){
137         max = valueNot;
138         min = valueNot - (valueNot - valueIS) * 2;
139     }
140     else
141     {
142         min = valueNot;
143         max = valueNot + (valueIS - valueNot) * 2;
144     }
145 
146     if (typeof shape === "undefined") shape = FuzzyShape.LINEAR;
147     if (typeof shapeAbove === "undefined") shapeAbove = shape;
148     if (typeof plateauMin === "undefined") plateauMin = FzMath.mean([min, max]);
149     if (typeof plateauMax === "undefined") plateauMax = FzMath.mean([min, max]);
150 
151     this.name = name;
152     this.min = min;
153     this.max = max;
154     this.shapeIn = shape;
155     this.shapeOut = shapeAbove;
156     var amplitude = max - min;
157     this.plateauMin = plateauMin;
158     this.plateauMax = plateauMax;
159 
160     this.algorithm = algorithm;
161 }

163 /**
164     * Checks if a value is contained in the set.
165     * @param {Number|FuzzyValue} v The value to test.
166     * @param {FuzzyQuantifier|Number} [quantifier=FuzzyQuantifier.NONE] Checks in which part of the set the value is in.
167     * @return {FuzzyVeracity} The veracity.
168     */
169 FuzzySet.prototype.contains = function ( v, quantifier )
170 {
171     var value;
172     if (v instanceof FuzzyValue) value = v.crispify(false);
173     else value = v;
174 
175     quantifier = FuzzyLogic.getQuantifier(quantifier);
176 
177     if (value >= this.plateauMin && value <= this.plateauMax)
178     {
179         return quantifier(1, this.algorithm);
180     }
181     else if (value < this.plateauMin)
182     {
183         if (this.shapeIn === FuzzyShape.CONSTANT)
184         {
185             return quantifier(1, this.algorithm);
186         }
187         else if (this.shapeIn === FuzzyShape.SQUARE)
188         {
189             var min = FzMath.mean(this.plateauMin, this.min);
190             if (value >= min) return quantifier(1, this.algorithm);
191             else return quantifier(0, this.algorithm);
192         }
193         else if (this.shapeIn === FuzzyShape.LINEAR)
194         {
195             if (value < this.min) return quantifier(0, this.algorithm);
196             else return quantifier( (value-this.min) / (this.plateauMin - this.min), this.algorithm );
197         }
198         else if (this.shapeIn === FuzzyShape.SIGMOID)
199         {
200             var mid = (this.plateauMin + this.min) / 2;
201             var rate = 6 / (this.plateauMin - this.min);
202             return quantifier(FzMath.logistic(value, mid, 0, 1, rate), this.algorithm);
203         }
204         else if (this.shapeIn === FuzzyShape.GAUSSIAN)
205         {
206             var width = this.plateauMin - this.min;
207             return quantifier( FzMath.gaussian( value, 0, 1, this.plateauMin, width) , this.algorithm);
208         }
209         else return quantifier(0, this.algorithm);
210     }
211     else
212     {
213         if (this.shapeOut === FuzzyShape.CONSTANT)
214         {
215             return quantifier(1, this.algorithm);
216         }
217         else if (this.shapeOut === FuzzyShape.SQUARE)
218         {
219             var max = FzMath.mean(this.plateauMax, this.max);
220             if (value <= max) return quantifier(1, this.algorithm);
221             else return quantifier(0, this.algorithm);
222         }
223         else if (this.shapeOut === FuzzyShape.LINEAR)
224         {
225             if (value > this.max) return quantifier(0, this.algorithm);
226             else return quantifier (1 - ((value - this.plateauMax ) / (this.max - this.plateauMax) ), this.algorithm);
227         }
228         else if (this.shapeOut === FuzzyShape.SIGMOID)
229         {
230             var mid = (this.plateauMax + this.max) / 2;
231             var rate = 6 / (this.max - this.plateauMax);
232             return quantifier( 1 - FzMath.logistic(value, mid, 0, 1, rate) , this.algorithm);
233         }
234         else if (this.shapeOut === FuzzyShape.GAUSSIAN)
235         {
236             var width = this.max - this.plateauMax;
237             return quantifier( FzMath.gaussian( value, 0, 1, this.plateauMax, width), this.algorithm );
238         }
239         else return quantifier(0, this.algorithm);
240     } 
241 }

243 /**
244     * Gets a list of precise values from the set corresponding to the given veracity.
245     * @param {FuzzyVeracity|Number} [veracity=0.5] The veracity
246     * @return {Number[]} The list of possible crisp values, ordered from min to max.
247     */
248 FuzzySet.prototype.getValues = function ( veracity )
249 {
250     if (typeof veracity === "undefined") veracity = 0.5;
251     if (veracity instanceof FuzzyVeracity) veracity = veracity.veracity;
252 
253     var defaultValue = FzMath.mean( [this.plateauMin, this.plateauMax] );
254 
255     if ( this.shapeIn === FuzzyShape.CONSTANT && this.shapeOut === FuzzyShape.CONSTANT ) return [ this.min, this.plateauMin, defaultValue, this.plateauMax, this.max];
256     
257     var crisp = [];
258     
259     if (veracity >= 1) crisp = [this.plateauMin, defaultValue, this.plateauMax];
260 
261     // below
262     if (this.shapeIn === FuzzyShape.CONSTANT && veracity == 1)
263     {
264         crisp.push(this.min);
265     }
266     else if (this.shapeIn === FuzzyShape.SQUARE)
267     {
268         if (veracity >= 0.5) crisp.push( this.plateauMin );
269         else crisp.push( this.min );
270     }
271     else if (this.shapeIn === FuzzyShape.LINEAR)
272     {
273         range = this.plateauMin - this.min;
274 
275         crisp.push( this.min + range * veracity );
276     }
277     else if (this.shapeIn === FuzzyShape.SIGMOID)
278     {
279         mid = (this.plateauMin + this.min) / 2;
280         crisp.push( FzMath.inverseLogistic(veracity, mid) );
281     }
282     else if (this.shapeIn === FuzzyShape.GAUSSIAN)
283     {
284         var width = this.plateauMin - this.min;
285         var g = FzMath.inverseGaussian( veracity, 0, 1, this.plateauMin, width);
286         crisp.push( g[0] );
287     }
288 
289     //above
290     if (this.shapeOut === FuzzyShape.CONSTANT && veracity == 1)
291     {
292         crisp.push(this.max);
293     }
294     if (this.shapeOut === FuzzyShape.SQUARE)
295     {
296         if (veracity >= 0.5) crisp.push( this.plateauMax );
297         else crisp.push( this.max );
298     }
299     else if (this.shapeOut === FuzzyShape.LINEAR)
300     {
301         range = this.max - this.plateauMax;
302 
303         crisp.push( this.max + 1 - (range * veracity) );
304     }
305     else if (this.shapeOut === FuzzyShape.SIGMOID)
306     {
307         mid = (this.plateauMax + this.max) / 2;
308         crisp.push( FzMath.inverseLogistic( 1-veracity, mid, 0, 1 ) );
309     }
310     else if (this.shapeOut === FuzzyShape.GAUSSIAN)
311     {
312         width = this.max - this.plateauMax;
313         var g = FzMath.inverseGaussian( 1-veracity, 0, 1, this.plateauMax, width);
314         crisp.push( g[1] );
315     }
316 
317     // Clamp
318     for(var i = 0, num = crisp.length; i < num; i++)
319     {
320         if ( crisp[i] > this.max ) crisp[i] = this.max;
321         if ( crisp[i] < this.min ) crisp[i] = this.min;
322     }
323 
324     return crisp.sort();
325 }

327 /**
328     * Gets a list of precise values from the set corresponding to the quantifier
329     * @param {FuzzyModifier} [quantifier=FuzzyModifier.AVERAGE] The quantifier
330     * @param {FuzzyVeracity|Number} [veracity=0.5] The veracity
331     * @return {Number[]} The list of possible crisp values, ordered from min to max.
332     */
333 FuzzySet.prototype.crispify = function ( quantifier, veracity )
334 {
335     quantifier = FuzzyLogic.getQuantifier(quantifier);
336     var v;
337     if (typeof veracity === "undefined") v = quantifier();
338     else if (veracity instanceof FuzzyVeracity) v = veracity.veracity;
339     else v = veracity;
340 
341     v = quantifier(v, this.algorithm, true).veracity;
342     return this.getValues( v );
343 }

345 /**
346  * Gets the closest quantifier to this value
347  * @param {Number|FuzzyValue} val The value to quantify
348  * @return {FuzzyQuantifier} The quantifier
349  */
350 FuzzySet.prototype.quantify = function ( value )
351 {
352     if (!(value instanceof FuzzyValue)) value = new FuzzyValue(value, "", this.algorithm);
353     var val = value.crispify(false);
354 
355     if ( this.shapeIn === FuzzyShape.CONSTANT && this.shapeOut === FuzzyShape.CONSTANT ) return FuzzyQuantifier.IS;
356     if (val < this.min && this.shapeIn === FuzzyShape.CONSTANT ) return FuzzyQuantifier.IS;
357     if (val < this.min) return FuzzyQuantifier.IS_NOT;
358     if (val > this.max && this.shapeOut === FuzzyShape.CONSTANT ) return FuzzyQuantifier.IS;
359     if (val > this.max) return FuzzyQuantifier.IS_NOT;
360     if (val >= this.plateauMin && val <= this.plateauMax) return FuzzyQuantifier.IS;
361 
362     var quantifier = FuzzyQuantifier.NOT;
363     var veracity = this.contains(val).veracity;
364 
365     var distance = 1;
366     for (var i in FuzzyQuantifier)
367     {
368         var q = FuzzyQuantifier[i];
369         var test = Math.abs( q() - veracity );
370         
371         if (test < distance)
372         {
373             distance = test;
374             quantifier = q;
375         }
376     }
377 
378     return quantifier;
379 }

392 /**
393  * Do not use the constructor of this class, use {@link FuzzyLogic.newSet} to create a new set.<br >
394  * @class
395  * @classdesc FuzzyValue is a value to be used with Fuzzy Logic. It's inclusion in a set can be tested with {@link FuzzyValue.IS} or {@link FuzzyValue.IS_NOT},<br />
396  * Use {@link FuzzyValue.SET} or {@link FuzzyLogic.THEN} to change this value using a {@link FuzzySet}
397  * @example
398  * var logic = new FuzzyLogic();
399  * var comfortable = logic.newSet("Warm", 17, 20);
400  * var temperature = logic.newValue( 18 );
401  * // Test if this temperature is comfortable
402  * logic.IF(
403  *    temperature.IS_NOT(comfortable);
404  * )
405  * // Set it very comfortable
406  * logic.THEN ( temperature.SET(comfortable, "Very"); )
407  * @property {string} unit The unit to display when returning this value as a string.
408  * @property {bool} reportEnabled Enables or disable report generation when crispifying. Disabled by default to improve performance.
409  * @property {string[][]} report The report (explanation) of the latest crispification {@link FuzzyValue.crispify}.<br />
410  * It is an Array containing Arrays of strings. Each sub-array is the report of one rule, which you can print with <code>.join(newLine)</code> for example.
411  */
412 function FuzzyValue( value, unit, algorithm, crispAlgorithm )
413 {
414     if (typeof unit === "undefined") unit = "";
415     if (typeof value === "undefined") value = 0;
416     this.value = value;
417     this.unit = unit;
418     this.sets = [];
419     this.algorithm = algorithm;
420     this.crispAlgorithm = crispAlgorithm;
421 
422     this.report = [];
423     this.reportEnabled = false;
424     this.numRules = 0;
425 }

427 /**
428  * Tests the inclusion of the value in the set
429  * @param {FuzzySet} set The set which may include the value.
430  * @param {FuzzyQuantifier|string} quantifier A quantifier.
431  * @return {FuzzyVeracity} The veracity of the inclusion of the value in the set.
432  */
433 FuzzyValue.prototype.IS = function(set, quantifier)
434 {
435     var v = set.contains( this, quantifier );
436     return v;
437 }

439 /**
440  * Tests the exclusion of the value in the set
441  * @param {FuzzySet} set The set which may (not) include the value.
442  * @param {FuzzyQuantifier|string} quantifier A quantifier.
443  * @return {FuzzyVeracity} The veracity of the exclusion of the value in the set.
444  */
445 FuzzyValue.prototype.IS_NOT = function (set, quantifier)
446 {
447     var x = set.contains( this.value, quantifier );
448     return x.NEGATE();
449 }

451 /**
452  * Changes the value according to a new veracity.
453  * @param {FuzzySet} set The set.
454  * @param {FuzzyModifier} [quantifier=FuzzyModifier.NONE] The quantifier
455  * @param {FuzzyVeracity} [v] The veracity.
456  */
457 FuzzyValue.prototype.SET = function ( set,  quantifier, v )
458 {
459     if (typeof v === "undefined") v = new FuzzyVeracity(1, this.algorithm);
460     
461     quantifier = FuzzyLogic.getQuantifier(quantifier);
462     
463     this.numRules++;
464     v.ruleNum = this.numRules;
465 
466     // Check if this set is already here
467     for (var i = 0, num = this.sets.length; i < num; i++)
468     {
469         if (set.name == this.sets[i].name) 
470         {
471             this.sets[i].quantifiers.push(quantifier);
472             this.sets[i].veracities.push(v);
473             return;
474         }
475     }
476 
477     //otherwise, add it
478     set.quantifiers = [quantifier];
479     set.veracities = [v];
480     this.sets.push( set );
481 }

483 /**
484  * Computes a crisp value depending on the inclusions which have been set before using {@link FuzzyValue.SET}.
485  * @param {bool} [clearSets=true] When crispifying, the sets added with {@link FuzzyValue.SET} are cleared, this means changes made before the call to crispify() will be lost in subsequent calls. Set this parameter to false to keep these previous changes.
486  * @param {FuzzyCrispAlgorithm} [algorithm] Change the algorithm to use for crispification.
487  * @return {Number} The crisp (i.e. standard) value.
488  */
489 FuzzyValue.prototype.crispify = function ( clearSets, algorithm )
490 {
491     if (typeof clearSets === "undefined") clearSets = true;
492     if (typeof algorithm === "undefined") algorithm = this.crispAlgorithm;
493 
494     if (this.sets.length == 0) return this.value;
495 
496     var crisp = 0;
497     this.report = [];
498 
499     function ruleSorter(a, b)
500     {
501         return a.number - b.number;
502     }
503 
504     // get all average values
505     // and veracities from the sets
506     var sumWeights = 0;
507     for (var i = 0, num = this.sets.length; i < num; i++)
508     {
509         var set = this.sets[i];
510         for( var j = 0, numV = set.veracities.length; j < numV; j++)
511         {
512             // the veracity
513             var v = set.veracities[j];
514             var q = set.quantifiers[j];
515             // the corresponding values
516             var vals = set.crispify( q, v );
517             var val;
518             var ver;
519 
520             if (algorithm == FuzzyCrispAlgorithm.CENTROID || algorithm == FuzzyCrispAlgorithm.MEAN) val = FzMath.mean(vals);
521             else if (algorithm == FuzzyCrispAlgorithm.CENTROID_LOWER || algorithm == FuzzyCrispAlgorithm.MEAN_LOWER) val = vals[0];
522             else if (algorithm == FuzzyCrispAlgorithm.CENTROID_HIGHER || algorithm == FuzzyCrispAlgorithm.MEAN_HIGHER) val = vals[vals.length-1];
523 
524             if (algorithm == FuzzyCrispAlgorithm.CENTROID || algorithm == FuzzyCrispAlgorithm.CENTROID_LOWER || algorithm == FuzzyCrispAlgorithm.CENTROID_HIGHER)
525             {
526                 crisp += val * v.veracity;
527                 ver = v.veracity;
528             }
529             else if (algorithm == FuzzyCrispAlgorithm.MEAN || algorithm == FuzzyCrispAlgorithm.MEAN_LOWER || algorithm == FuzzyCrispAlgorithm.MEAN_HIGHER)
530             {
531                 crisp += val;
532                 ver = 1;
533             }
534 
535             sumWeights += ver;
536             
537 
538             // generate report
539             if (this.reportEnabled)
540             {
541                 for (var iVals = 0, numVals = vals.length; iVals < numVals; iVals++)
542                 {
543                     vals[iVals] = Math.round(vals[iVals]*1000)/1000;
544                 }
545 
546                 var reportRule = [];
547                 reportRule.push( "Rule #" + v.ruleNum +": Set " + set.toString() + " (" + q.toString() + ")" );
548                 reportRule.push( "Gives value: " + Math.round(val*1000)/1000 + " from these values: [ " + vals.join(", ") + " ]");
549                 reportRule.push( "With a veracity of: " + Math.round(ver*1000)/1000 );
550                 reportRule.number = v.ruleNum;
551                 this.report.push( reportRule );
552             }
553         }
554     }
555             
556     if (sumWeights != 0) crisp = crisp / sumWeights;
557 
558 
559     //sort the report
560     if (this.reportEnabled) this.report.sort(ruleSorter);
561 
562     if (clearSets)
563     {
564         // freeze all
565         this.value = crisp;
566         //reset sets
567         this.sets = [];
568     }
569 
570     return crisp;
571 }

599 /**
600  * Returns a string representation of the value with its unit (if any)
601  * @param {FuzzyCrispAlgorithm} [crispAlgorithm] The algorithm to use for crispification before returning the string. Uses the algorithm set when creating the {@link FuzzyLogic} instance by default.
602  * @param {FuzzySet} [set] A set to quantify the value
603  * @return {string} The description of the value
604  */
605 FuzzyValue.prototype.toString = function( crispAlgorithm, set )
606 {
607     var v = this.crispify( false, crispAlgorithm );
608     v = Math.round( v * 100 ) / 100;
609     var str = v + this.unit;
610     if (typeof set !== "undefined") str += " is " + set.quantify(this).toString() + " " + set.toString();
611     return str;
612 }

616 /**
617  * @class
618  * @classdesc FuzzyVeracity is the fuzzy equivalent of a crisp boolean in boolean logic. It represents the result of logical operations (IS, AND, OR...).<br />
619  * Its value can vary in the range [0.0, 1.0], 0.0 being the equivalent of "false" and 1.0 of "true".<br />
620  * You can acces logic operators through instances of this class, which enable fluent syntax for rules (Except IS and IS_NOT which are members of {@link FuzzyValue}).<br />
621  * Operators are methods in upper case.
622  * @example
623  * var logic = new FuzzyLogic();
624  * var intense = logic.newSet("Intense", 0, 255);
625  * // An RGB color to test
626  * var color = [255,200,10];
627  * // Separate channels
628  * var redChannel = logic.newValue( color[0] );
629  * var greenChannel = logic.newValue( color[1] );
630  * var blueChannel = logic.newValue( color[2] );
631  * // isRed will be a FuzzyVeracity, the result of the test.
632  * // Note that FuzzyValue.IS returns a FuzzyVeracity, on which the methods AND and NOR are called.
633  * // But this is all internal and you don't really need to know that to use this syntax.
634  * var isRed = redChannel.IS( intense )
635  *      .AND(
636  *         greenChannel.IS( intense ).
637  *         NOR( blueChannel.IS( intense ))
638  *      )
639  * @property {Number} veracity The veracity level in the range [0.0, 1.0]
640  */
641 function FuzzyVeracity( veracity, algorithm )
642 {
643     if (typeof above === "undefined") above = false;
644     if (typeof algorithm === "undefined") algorithm = FuzzyLogicAlgorithm.LINEAR;
645     this.veracity = veracity;
646     this.algorithm = algorithm;
647 }

658 /**
659  * The equivalent of the boolean operation <code>this && other</code>
660  * @param {FuzzyVeracity} other The other operand.
661  * @return {FuzzyVeracity}
662  */
663 FuzzyVeracity.prototype.AND = function( other )
664 {
665     var x = this.veracity;
666     var y = other.veracity;
667 
668     var v = 0;
669     if (this.algorithm == FuzzyLogicAlgorithm.LINEAR) v = Math.min(x, y);
670     else v = x*y;
671 
672     return new FuzzyVeracity( v, this.algorithm );

677  * @param {FuzzyVeracity} other The other operand.

683     var y = other.veracity;
